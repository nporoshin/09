методы списков(list)
.append(x)   - добавление 'элемента в конец
.insert(i,x) - вставка на i-ое место
.extend(L)   - добавление в конец списка L
.sort()      - сортировка
.remove(x)   - удаляет первый x

методы словарей (dict)
.get(key[, default]) - возвращает значение ключа,
                       но если его нет возвращает default (по умолчанию None).
.items()             - возвращает пары (ключ, значение)
.keys()              - возвращает ключи в словаре
.values()            - возвращает значения в словаре
.pop(key[, default]) - удаляет ключ и возвращает значение
                       Если ключа нет, возвращает default (по умолчанию бросает исключение)


методы множеств (set)
.union(other, ...) или  (|)         - объединение нескольких множеств (other, ...).
.intersection(other, ...) или  (&)  - пересечение нескольких множеств (other, ...).
.difference(other, ...) или  (-)    - множество из всех элементов исходного множества,
                                      не принадлежащие ни одному из other
.update(other, ...) или (|=)        - объединение исходного множества с other
.add(элемент)                       - добавляет элемент в множество

методы строк (str)
.split(разделитель)                  - Разбиение строки по разделителю
.isdigit()                           - Состоит ли строка из цифр
.isalpha()                           - Состоит ли строка из букв
.isalnum()                           - Состоит ли строка из цифр или букв
.replace(шаблон, замена[, maxcount]) - Замена шаблона на замену.
                                       maxcount ограничивает количество замен
